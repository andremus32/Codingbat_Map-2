//Map-2 > wordAppend
public String wordAppend(String[] strings) {
  int count = 0;
  String result = "";
  Map<Integer, Integer> map = new HashMap<>();
  
  for(int i = 0; i < strings.length; i++){
    for(int j = 0; j <= i; j++){
      if(strings[i].equals(strings[j])){
        count++;
      } 
    }
    map.put(i, count);
    count = 0;
  }
  
  for(int i = 0; i < strings.length; i++){
    if(map.get(i) > 0 && map.get(i) % 2 == 0){
      result += strings[i];
    }
  }
  return result;
}
